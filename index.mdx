    Why Killa?
  </h2>

<CardGroup cols={3}>
  <Card title="Lightweight" icon="feather">
    Zero bloat. Killa adds minimal overhead to your bundle with no unnecessary abstractions.
  </Card>
  <Card title="Immutable State" icon="lock">
    Internal state is always immutable. Deep equality checks prevent unnecessary re-renders.
  </Card>
  <Card title="Selective Subscriptions" icon="filter">
    Subscribe only to the state slices you need. Components re-render only when their selected state changes.
  </Card>
  <Card title="React and Vanilla" icon="code">
    Works with plain JavaScript and React. One API for both environments.
  </Card>
  <Card title="Persist Middleware" icon="database">
    Built-in persistence with localStorage, sessionStorage, or any custom storage backend.
  </Card>
  <Card title="TypeScript First" icon="brackets-curly">

<div style={{marginBottom: "64px"}}>
  <h2 style={{fontSize: "38px", fontWeight: "700", color: "#ffffff", textAlign: "center", marginBottom: "40px"}}>
    Quick example
  </h2>

<Tabs>
  <Tab title="Vanilla JS">
    ```js
    import { createStore } from 'killa'

    const store = createStore({ counter: 0 })

    store.subscribe((state, prevState) => {
      console.log('State changed:', state.counter)
    }, (state) => state.counter)

    store.setState((state) => ({
      ...state,
      counter: state.counter + 1
    }))

    store.getState() // { counter: 1 }
    ```
  </Tab>
  <Tab title="React">
    ```jsx
    import { createStore } from 'killa'
    import { useStore } from 'killa/react'

    const store = createStore({ counter: 0 })

    export function Counter() {
      const [state, setState] = useStore(store, (s) => ({
        counter: s.counter
      }))

      return (
        <div>
          <p>Counter: {state.counter}</p>
          <button onClick={() => setState((s) => ({ ...s, counter: s.counter + 1 }))}>
            +1
          </button>
        </div>
      )
    }
    ```
  </Tab>
  <Tab title="With Actions">
    ```js
    import { createStore } from 'killa'

    const store = createStore((get, set) => ({
      count: 0,
      inc: () => set(() => ({ count: get().count + 1 })),
      dec: () => set(() => ({ count: get().count - 1 })),
      reset: () => set(() => ({ count: 0 }))
    }))

    store.getState().inc()
    store.getState().inc()
    store.getState().count // 2
    ```
  </Tab>
</Tabs>
